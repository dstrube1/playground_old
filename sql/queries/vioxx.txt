SELECT dbo_p_Patient.last_name AS Last_name, dbo_p_Patient.first_name AS First_name, dbo_p_Patient.billing_id AS Billing_id, dbo_p_Treatment_Item.treatment_description AS Treatment_description, user_short_name AS [User], dbo_p_Treatment_Item.begin_date AS Begin_date, dbo_p_Treatment_Item.end_date AS End_date, dbo_p_Treatment_Item.treatment_status AS Treatment_status
FROM (dbo_p_Patient INNER JOIN dbo_p_Treatment_Item ON dbo_p_Patient.cpr_id = dbo_p_Treatment_Item.cpr_id) INNER JOIN dbo_c_User ON dbo_p_Treatment_Item.created_by = dbo_c_User.user_id
WHERE (((dbo_p_Treatment_Item.treatment_description) Like "%viox%"));
