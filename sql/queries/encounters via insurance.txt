SELECT dbo_p_Patient_Insurance.insurance_id, dbo_c_Insurance_Type.insurance_type, dbo_p_Patient_Encounter.encounter_type, dbo_p_Patient_Encounter.indirect_flag, dbo_p_Patient_Encounter.cpr_id, dbo_p_Patient_Encounter.encounter_date, dbo_p_Patient_Encounter.created_by
FROM ((dbo_p_Patient_Insurance INNER JOIN (dbo_p_Patient INNER JOIN dbo_p_Patient_Encounter ON dbo_p_Patient.cpr_id = dbo_p_Patient_Encounter.cpr_id) ON dbo_p_Patient_Insurance.cpr_id = dbo_p_Patient_Encounter.cpr_id) INNER JOIN dbo_c_Insurance ON dbo_p_Patient_Insurance.insurance_id = dbo_c_Insurance.insurance_id) INNER JOIN dbo_c_Insurance_Type ON dbo_c_Insurance.insurance_type = dbo_c_Insurance_Type.insurance_type
WHERE (((dbo_p_Patient_Encounter.indirect_flag)="d") AND ((dbo_p_Patient_Encounter.encounter_date)>=#8/1/2003# And (dbo_p_Patient_Encounter.encounter_date)<=#8/31/2003#));
