SELECT DISTINCT [dbo_c_user].[user_full_name] AS [User], (select count(*) from dbo_p_Patient_Encounter  where dbo_p_Patient_Encounter.encounter_type not in ('PHONE') and dbo_c_user.user_id = dbo_p_Patient_Encounter.attending_doctor and ([indirect_flag]="D") And ([encounter_date]>=begin) And ([encounter_date]<end)) AS [Count]
FROM dbo_c_user
WHERE user_status = 'OK' and dbo_c_user.user_id = userid;
