SELECT distinct [dbo_p_Patient].[last_name], [dbo_p_Patient].[first_name], [dbo_p_Patient].[date_of_birth], [dbo_p_Patient].[billing_id] 
FROM ((dbo_c_Assessment_Definition INNER JOIN dbo_p_Assessment ON dbo_c_Assessment_Definition.assessment_id = dbo_p_Assessment.assessment_id) INNER JOIN dbo_p_Patient ON dbo_p_Assessment.cpr_id = dbo_p_Patient.cpr_id) INNER JOIN dbo_p_Treatment_Item ON dbo_p_Patient.cpr_id = dbo_p_Treatment_Item.cpr_id
WHERE icd_9_code In ("314.01","314.00") and  [dbo_p_Treatment_Item].[treatment_description] like "Adderal*";
